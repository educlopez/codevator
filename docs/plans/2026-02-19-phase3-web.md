> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build an interactive scroll-driven landing page where scrolling simulates riding an elevator, with Severance/Lumon Industries aesthetic, playable sound demos, and a terminal simulation.

**Architecture:** Next.js 15 App Router with static export. GSAP ScrollTrigger manages scroll-driven animations with pinning for each "floor". Each floor is a full-viewport section. **Tech Stack:** Next.js 15, Tailwind CSS v4, GSAP + ScrollTrigger, Web Audio API, Framer Motion (micro-interactions)

**Step 2: Create ElevatorShell component**

This manages the GSAP scroll setup and floor tracking.

Create `packages/web/src/components/ElevatorShell.tsx`:
```tsx
"use client";

import { useEffect, useRef, useState } from "react";
import { gsap } from "gsap";
import { ScrollTrigger } from "gsap/ScrollTrigger";
import { FloorIndicator } from "./FloorIndicator";

gsap.registerPlugin(ScrollTrigger);

const FLOORS = ["B", "1", "2", "3", "4", "5"];

interface ElevatorShellProps {
  children: React.ReactNode[];
}

export function ElevatorShell({ children }: ElevatorShellProps) {
  const [currentFloor, setCurrentFloor] = useState(0);
  const containerRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const sections = containerRef.current?.querySelectorAll(".floor-section");
    if (!sections) return;

    const triggers: ScrollTrigger[] = [];

    sections.forEach((section, i) => {
      const trigger = ScrollTrigger.create({
        trigger: section,
        start: "top center",
        end: "bottom center",
        onEnter: () => setCurrentFloor(i),
        onEnterBack: () => setCurrentFloor(i),
      });
      triggers.push(trigger);
    });

    return () => {
      triggers.forEach((t) => t.kill());
    };
  }, []);

  return (
    <>
      <FloorIndicator currentFloor={currentFloor} floors={FLOORS} />
      <div ref={containerRef}>
        {children.map((child, i) => (
          <section
            key={i}
            className="floor-section min-h-screen relative flex items-center justify-center"
          >
            {child}
          </section>
        ))}
      </div>
    </>
  );
}
```

```tsx
"use client";

import { useEffect, useRef } from "react";
import { gsap } from "gsap";
import { ScrollTrigger } from "gsap/ScrollTrigger";

gsap.registerPlugin(ScrollTrigger);

export function ElevatorDoors() {
  const containerRef = useRef<HTMLDivElement>(null);
  const leftDoorRef = useRef<HTMLDivElement>(null);
  const rightDoorRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const ctx = gsap.context(() => {
      const tl = gsap.timeline({
        scrollTrigger: {
          trigger: containerRef.current,
          start: "top top",
          end: "bottom top",
          scrub: 1,
          pin: true,
          pinSpacing: true,
        },
      });

      tl.to(leftDoorRef.current, { xPercent: -100, duration: 1 }, 0);
      tl.to(rightDoorRef.current, { xPercent: 100, duration: 1 }, 0);
    }, containerRef);

    return () => ctx.revert();
  }, []);

  return (
    <div ref={containerRef} className="relative w-full h-screen overflow-hidden bg-lumon-dark">

      <div className="absolute inset-0 flex flex-col items-center justify-center bg-lumon-bg">
```

### Task 5: Web Audio Engine

**Files:**
- Create: `packages/web/src/lib/audio.ts`

We synthesize all sounds in the browser using Web Audio API — no mp3 files needed.

**Step 1: Create audio engine**

Create `packages/web/src/lib/audio.ts`:
```ts
type Mode = "elevator" | "typewriter" | "ambient" | "retro" | "minimal";

let audioCtx: AudioContext | null = null;
let activeNodes: AudioNode[] = [];
let currentMode: Mode | null = null;

function getCtx(): AudioContext {
  if (!audioCtx) {
    audioCtx = new AudioContext();
  }
  if (audioCtx.state === "suspended") {
    audioCtx.resume();
  }
  return audioCtx;
}

function stopAll() {
  activeNodes.forEach((node) => {
    try {
      if (node instanceof OscillatorNode) node.stop();
      if (node instanceof AudioBufferSourceNode) node.stop();
      node.disconnect();
    } catch {}
  });
  activeNodes = [];
  currentMode = null;
}

function playElevator() {
  const ctx = getCtx();
  const gain = ctx.createGain();
  gain.gain.value = 0.15;
  gain.connect(ctx.destination);

  // Warm chord: C major 7 with slight detuning for warmth
  const frequencies = [261.63, 329.63, 392.0, 493.88];
  frequencies.forEach((freq) => {
    const osc = ctx.createOscillator();
    osc.type = "sine";
    osc.frequency.value = freq + (Math.random() - 0.5) * 2;

    // Slight tremolo for "muzak" feel
    const tremolo = ctx.createGain();
    const lfo = ctx.createOscillator();
    lfo.type = "sine";
    lfo.frequency.value = 0.5 + Math.random() * 0.3;
    const lfoGain = ctx.createGain();
    lfoGain.gain.value = 0.1;
    lfo.connect(lfoGain);
    lfoGain.connect(tremolo.gain);
    tremolo.gain.value = 0.8;

    osc.connect(tremolo);
    tremolo.connect(gain);
    osc.start();
    lfo.start();

    activeNodes.push(osc, lfo);
  });

  activeNodes.push(gain);
}

function playTypewriter() {
  const ctx = getCtx();
  const gain = ctx.createGain();
  gain.gain.value = 0.3;
  gain.connect(ctx.destination);
  activeNodes.push(gain);

  function tick() {
    if (currentMode !== "typewriter") return;

    const bufferSize = ctx.sampleRate * 0.02;
    const buffer = ctx.createBuffer(1, bufferSize, ctx.sampleRate);
    const data = buffer.getChannelData(0);
    for (let i = 0; i < bufferSize; i++) {
      data[i] = (Math.random() * 2 - 1) * Math.exp(-i / (bufferSize * 0.1));
    }

    const source = ctx.createBufferSource();
    source.buffer = buffer;

    const filter = ctx.createBiquadFilter();
    filter.type = "highpass";
    filter.frequency.value = 800 + Math.random() * 2000;

    source.connect(filter);
    filter.connect(gain);
    source.start();

    const nextDelay = 60 + Math.random() * 140;
    setTimeout(tick, nextDelay);
  }

  tick();
}

function playAmbient() {
  const ctx = getCtx();

  // Pink noise for rain
  const bufferSize = 2 * ctx.sampleRate;
  const buffer = ctx.createBuffer(1, bufferSize, ctx.sampleRate);
  const data = buffer.getChannelData(0);

  let b0 = 0, b1 = 0, b2 = 0, b3 = 0, b4 = 0, b5 = 0, b6 = 0;
  for (let i = 0; i < bufferSize; i++) {
    const white = Math.random() * 2 - 1;
    b0 = 0.99886 * b0 + white * 0.0555179;
    b1 = 0.99332 * b1 + white * 0.0750759;
    b2 = 0.96900 * b2 + white * 0.1538520;
    b3 = 0.86650 * b3 + white * 0.3104856;
    b4 = 0.55000 * b4 + white * 0.5329522;
    b5 = -0.7616 * b5 - white * 0.0168980;
    data[i] = (b0 + b1 + b2 + b3 + b4 + b5 + b6 + white * 0.5362) * 0.05;
    b6 = white * 0.115926;
  }

  const source = ctx.createBufferSource();
  source.buffer = buffer;
  source.loop = true;

  const filter = ctx.createBiquadFilter();
  filter.type = "lowpass";
  filter.frequency.value = 600;

  const gain = ctx.createGain();
  gain.gain.value = 0.4;

  source.connect(filter);
  filter.connect(gain);
  gain.connect(ctx.destination);
  source.start();

  // Low drone
  const drone = ctx.createOscillator();
  drone.type = "sine";
  drone.frequency.value = 174;
  const droneGain = ctx.createGain();
  droneGain.gain.value = 0.04;
  drone.connect(droneGain);
  droneGain.connect(ctx.destination);
  drone.start();

  activeNodes.push(source, gain, drone, droneGain);
}

function playRetro() {
  const ctx = getCtx();
  const gain = ctx.createGain();
  gain.gain.value = 0.15;
  gain.connect(ctx.destination);
  activeNodes.push(gain);

  const notes = [523, 659, 784, 1047, 784, 659];
  let noteIndex = 0;

  function playNote() {
    if (currentMode !== "retro") return;

    const osc = ctx.createOscillator();
    osc.type = "square";
    osc.frequency.value = notes[noteIndex % notes.length];

    const noteGain = ctx.createGain();
    noteGain.gain.setValueAtTime(0.3, ctx.currentTime);
    noteGain.gain.exponentialRampToValueAtTime(0.01, ctx.currentTime + 0.14);

    osc.connect(noteGain);
    noteGain.connect(gain);
    osc.start();
    osc.stop(ctx.currentTime + 0.15);

    noteIndex++;
    setTimeout(playNote, 180);
  }

  playNote();
}

function playMinimal() {
  const ctx = getCtx();
  const osc = ctx.createOscillator();
  osc.type = "sine";
  osc.frequency.value = 60;

  const tremolo = ctx.createGain();
  const lfo = ctx.createOscillator();
  lfo.type = "sine";
  lfo.frequency.value = 1;
  const lfoGain = ctx.createGain();
  lfoGain.gain.value = 0.4;
  lfo.connect(lfoGain);
  lfoGain.connect(tremolo.gain);
  tremolo.gain.value = 0.5;

  const masterGain = ctx.createGain();
  masterGain.gain.value = 0.08;

  osc.connect(tremolo);
  tremolo.connect(masterGain);
  masterGain.connect(ctx.destination);
  osc.start();
  lfo.start();

  activeNodes.push(osc, lfo, masterGain);
}

export function playMode(mode: Mode) {
  if (currentMode === mode) return;
  stopAll();
  currentMode = mode;

  switch (mode) {
    case "elevator": return playElevator();
    case "typewriter": return playTypewriter();
    case "ambient": return playAmbient();
    case "retro": return playRetro();
    case "minimal": return playMinimal();
  }
}

export function stopAudio() {
  stopAll();
}

export function isAudioPlaying(): boolean {
  return currentMode !== null;
}

export function getCurrentMode(): Mode | null {
  return currentMode;
}
```

```tsx
"use client";

import { useEffect, useRef } from "react";

interface AudioVisualizerProps {
  active: boolean;
  color: string;
}

export function AudioVisualizer({ active, color }: AudioVisualizerProps) {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const animRef = useRef<number>();

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;

    const ctx = canvas.getContext("2d");
    if (!ctx) return;

    const bars = 32;
    const heights = new Float32Array(bars);

    function draw() {
      if (!ctx || !canvas) return;

      ctx.clearRect(0, 0, canvas.width, canvas.height);

      const barWidth = canvas.width / bars;
      const maxHeight = canvas.height * 0.8;

      for (let i = 0; i < bars; i++) {
        if (active) {
          // Smooth random movement
          const target = 0.2 + Math.random() * 0.8;
          heights[i] += (target - heights[i]) * 0.15;
        } else {
          heights[i] *= 0.95;
        }

        const h = heights[i] * maxHeight;
        const x = i * barWidth + barWidth * 0.15;
        const w = barWidth * 0.7;
        const y = (canvas.height - h) / 2;

        ctx.fillStyle = color + "80"; // 50% opacity
        ctx.beginPath();
        ctx.roundRect(x, y, w, h, 2);
        ctx.fill();
      }

      animRef.current = requestAnimationFrame(draw);
    }

    draw();

    return () => {
      if (animRef.current) cancelAnimationFrame(animRef.current);
    };
  }, [active, color]);

  return (
    <canvas
      ref={canvasRef}
      width={400}
      height={120}
      className="w-full max-w-md h-24 opacity-60"
    />
  );
}
```
